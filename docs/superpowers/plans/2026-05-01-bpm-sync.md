**Goal:** Make each deck's `Sync` button short-press to match the opposite deck BPM and long-press to restore that deck to its original BPM.

**Architecture:** Keep the BPM math and press-duration classification in `src/sync.js` so the behavior stays testable outside React. Wire the existing `Sync` buttons in `src/App.tsx` to pointer events for short press and long press, while preserving a keyboard-accessible click fallback.

**Tech Stack:** React 19, TypeScript/JS modules, Vite, node `assert`

const roundRate = (value) => Number(value.toFixed(3));

export const SYNC_LONG_PRESS_MS = 500;

export const getSyncPressAction = ({
  durationMs,
  longPressThresholdMs = SYNC_LONG_PRESS_MS,
}) => (durationMs >= longPressThresholdMs ? 'restore' : 'sync');

export const getSyncedPlaybackRate = ({
  sourceBpm,
  targetBaseBpm,
}) => {
  if (
    !Number.isFinite(sourceBpm) ||
    sourceBpm <= 0 ||
    !Number.isFinite(targetBaseBpm) ||
    targetBaseBpm <= 0
  ) {
    return null;
  }

  return roundRate(sourceBpm / targetBaseBpm);
};
```

const syncPressTimeoutRef = useRef<{ A: number | null; B: number | null }>({ A: null, B: null });
const syncPressStartedAtRef = useRef<{ A: number; B: number }>({ A: 0, B: 0 });
const syncLongPressTriggeredRef = useRef<{ A: boolean; B: boolean }>({ A: false, B: false });
const syncSuppressClickRef = useRef<{ A: boolean; B: boolean }>({ A: false, B: false });

const restoreDeckBpm = (deck: 'A' | 'B') => {
  const track = deck === 'A' ? trackA : trackB;

  if (track == null) {
    return;
  }

  if (deck === 'A') {
    setPlaybackRateA(1);
  } else {
    setPlaybackRateB(1);
  }
};

const clearSyncPressTimeout = (deck: 'A' | 'B') => {
  const timeoutId = syncPressTimeoutRef.current[deck];

  if (timeoutId != null) {
    window.clearTimeout(timeoutId);
    syncPressTimeoutRef.current[deck] = null;
  }
};

const handleSyncPointerDown = (deck: 'A' | 'B') => {
  syncPressStartedAtRef.current[deck] = Date.now();
  syncLongPressTriggeredRef.current[deck] = false;

  syncPressTimeoutRef.current[deck] = window.setTimeout(() => {
    syncLongPressTriggeredRef.current[deck] = true;
    syncSuppressClickRef.current[deck] = true;
    restoreDeckBpm(deck);
  }, SYNC_LONG_PRESS_MS);
};

const handleSyncPointerUp = (deck: 'A' | 'B') => {
  const durationMs = Date.now() - syncPressStartedAtRef.current[deck];
  const action = getSyncPressAction({ durationMs });

  clearSyncPressTimeout(deck);
  syncSuppressClickRef.current[deck] = true;

  if (action === 'restore' || syncLongPressTriggeredRef.current[deck]) {
    syncLongPressTriggeredRef.current[deck] = false;
    return;
  }

  syncDeckToOther(deck);
};
```
